lida_ongoing_sex_effect = {
	# stamina reduction on both parties
	if = {
		limit = {
			$STAMINA_COST_1$ > 0
		}
		save_scope_value_as = {
			name = root_stamina
			value = {
				add = scope:root_stamina
				subtract = $STAMINA_COST_1$
			}
		}
	}
	if = {
		limit = {
			$STAMINA_COST_2$ > 0
		}
		save_scope_value_as = {
			name = partner_stamina
			value = {
				add = scope:partner_stamina
				subtract = $STAMINA_COST_2$
			}
		}
	}
}

check_if_root_cum_effect = {
	if = {
		limit = {
			scope:root_stamina <= 0
		}
		save_scope_value_as = {
			name = root_cum
			value = yes
		}
		save_scope_value_as = {
			name = root_stamina
			value = max_sex_stamina
		}
		change_subdom_effect = { CHANGE = -2 }
	}
	else = {
		save_scope_value_as = {
			name = root_cum
			value = no
		}
	}
}
