become_more_sub_effect = {
	if = {
		limit = {
			NOT = {has_trait = lida_dom_track}
			has_trait = lida_sub_track
		}
		add_trait_xp = {
			trait = lida_sub_track
			value = $AMOUNT$
		}
	}
	else_if = {
		limit = {
			NOT = {
				has_trait = lida_dom_track
				has_trait = lida_sub_track
			}
		}
		add_trait = lida_sub_track
	}
	else_if = {
		limit = {
			has_trait = lida_dom_track
			has_trait_xp = {
				trait = lida_dom_track
				value <= 0
			}
		}
		remove_trait = lida_dom_track
	}
	else = {
		add_trait_xp = {
			trait = lida_dom_track
			value = -$AMOUNT$
		}
	}
}
